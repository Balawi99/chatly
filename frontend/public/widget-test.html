How to Integrate the Widget

To add the Chatly widget to your website, copy and paste the following code snippet just before the closing </body> tag:

<script>
  (function(w, d, s, o, f, js, fjs) {
    w['Chatly-Widget'] = o;
    w[o] = w[o] || function() {
      (w[o].q = w[o].q || []).push(arguments);
    };
    js = d.createElement(s);
    fjs = d.getElementsByTagName(s)[0];
    js.id = o;
    js.src = f;
    js.async = 1;
    fjs.parentNode.insertBefore(js, fjs);
  }(window, document, 'script', 'chatly', 'http://localhost:3001/widget.js'));
  chatly('init', { widgetId: 'YOUR_WIDGET_ID' });
</script>

Replace YOUR_WIDGET_ID with the ID of your widget from the Chatly dashboard.
